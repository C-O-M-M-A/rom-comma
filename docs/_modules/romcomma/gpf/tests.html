[docs]def kernel():
    lengthscales = [0.01 * np.ones(3), 0.03 * np.ones(3)]
    variance = 0.5 * np.eye(2)
    return kernels.RBF(variance, lengthscales)




[docs]def likelihood():
    variance = 0.0001 * np.eye(2)
    return likelihoods.MOGaussian(variance)
